  SELECT name, breed, origin
  FROM cats
  WHERE breed = 'Burmese';

SELECT results_eq (
  'basic_select',
  $$VALUES('Kitty'::text, 'Burmese'::text, 'TH'::text)$$,
  'should select all cats of Burmese breed'
);
SELECT * FROM finish();
